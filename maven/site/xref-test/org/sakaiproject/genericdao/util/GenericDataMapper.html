17  import java.lang.reflect.Type;

19  import org.sakaiproject.genericdao.api.mappers.DataMapper;
20  import org.sakaiproject.genericdao.util.TypeReference;

22  /***
23   * Allows a developer to use generics to define the persistent type when working with
24   * datamappers, you must run the empty constructor when constructing your class or this
25   * will fail to work, if you are not defining any constructors you do not have to do anything,
26   * if you decide to define some constructors then you must call super() in your constructor
27   * 
28   * @author Aaron Zeckoski (azeckoski @ gmail.com)
29   */
30  public abstract class GenericDataMapper<T> implements DataMapper {
31  
32      private GenericType genTypeRef = new GenericType();
33      private TypeReference<T> typeRef = new TypeReference<T>() {};
34  
35      /* (non-Javadoc)
36       * @see org.sakaiproject.genericdao.api.mappers.DataMapper#getPersistentType()
37       */
38      public Class<?> getPersistentType() {
39          @SuppressWarnings("unused")
40          Type t1 = typeRef.getType();
41          Type t = genTypeRef.getType();
42          return (Class<?>) t;
43      }
44  
45      protected class GenericType extends TypeReference<T> {}
46  
47  }
