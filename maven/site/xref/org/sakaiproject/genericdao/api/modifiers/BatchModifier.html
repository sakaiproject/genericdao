15  package org.sakaiproject.genericdao.api.modifiers;
16  
17  import java.io.Serializable;
18  import java.util.Set;
19  
20  /***
21   * This Modifier provides methods to modify groups of persistent objects
22   * within a single transaction (i.e. all operations have to succeed or
23   * the batch operation will fail)
24   * 
25   * @author Aaron Zeckoski (aaronz@vt.edu)
26   */
27  public interface BatchModifier {
28  
29  	/***
30  	 * Saves (creates or updates) a set of objects (persistent or transient) of
31  	 * the same class type in a single transaction<br/>
32  	 * <b>Warning:</b> Make sure all objects are of the same type and that
33  	 * the type is set as a persistent class for this DAO
34  	 * 
35  	 * @param entities a List of 0 or more objects to save
36  	 * (recommend use of HashSet)
37  	 */
38  	public <T> void saveSet(Set<T> entities);
39  
40  	/***
41  	 * This allows us to save a set of entities (objects) that are not 
42  	 * all of the same class type within a single transaction<br/>
43  	 * <b>Warning:</b> Make sure all objects are of the types that
44  	 * are set as persistent classes for this DAO and that each
45  	 * Set contains objects of the same type
46  	 * 
47  	 * @param entitySets an array of Sets of persistent objects (entities)
48  	 * (each Set should have one type of entity, recommend using HashSet)
49  	 */
50  	@SuppressWarnings("unchecked")
51     public void saveMixedSet(Set[] entitySets);
52  
53     /***
54      * Deletes (removes) a set of persistent objects in a single transaction
55      * by their unique ids
56      * 
57      * @param entityClass class type of the persistent object
58      * @param ids the identifiers of the objects to delete
59      */
60     public <T> void deleteSet(Class<T> entityClass, Serializable[] ids);
61  
62  	/***
63  	 * Deletes (removes) a set of objects (persistent or transient) of
64  	 * the same class type in a single transaction<br/>
65  	 * <b>Warning:</b> Make sure all objects are of the same type and that
66  	 * the type is set as a persistent class for this DAO
67  	 * 
68  	 * @param entities a Set of 0 or more objects to delete
69  	 * (recommend use of HashSet)
70  	 */
71  	public <T> void deleteSet(Set<T> entities);
72  
73  	/***
74  	 * This allows us to delete a set of entities (objects) that are not 
75  	 * all of the same class type within a single transaction<br/>
76  	 * <b>Warning:</b> Make sure all objects are of the types that
77  	 * are set as persistent classes for this DAO and that each
78  	 * Set contains objects of the same type
79  	 * 
80  	 * @param entitySets an array of Sets of persistent objects (entities)
81  	 * (each Set should have one type of entity, recommend using HashSet)
82  	 */
83  	@SuppressWarnings("unchecked")
84     public void deleteMixedSet(Set[] entitySets);
85  
86  }
