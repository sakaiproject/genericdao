19   * thrown if an object cannot be found in a cache

23  public class CacheKeyNotFoundException extends RuntimeException {
24  	private static final long serialVersionUID = 1L;
25  
26  	public String cacheName;
27  	public String cacheKey;
28  
29  	public CacheKeyNotFoundException(String message, String cacheName, String cacheKey) {
30  		super(message);
31  		this.cacheName = cacheName;
32  		this.cacheKey = cacheKey;
33  	}
34  
35  	public CacheKeyNotFoundException(String message, String cacheName,
36  			String cacheKey, Throwable cause) {

39  		this.cacheKey = cacheKey;
40  	}

42  }
