17  import java.lang.annotation.Documented;
18  import java.lang.annotation.ElementType;
19  import java.lang.annotation.Retention;
20  import java.lang.annotation.RetentionPolicy;
21  import java.lang.annotation.Target;
22  
23  
24  /***
25   * Indicates that this column represents a foreign key value which points
26   * to another persistent object, if this is a foreign key in the database
27   * to a non-persistent table then use {@link PersistentTransient} to indicate it should be ignored<br/>
28   * You can optionally specify a foreign table name and foreign id column manually if desired<br/>
29   * Should be placed on a field/method in a persistent class<br/>
30   * 
31   * @author Aaron Zeckoski (azeckoski @ gmail.com)
32   */
33  @Documented
34  @Retention(RetentionPolicy.RUNTIME)
35  @Target({ElementType.METHOD, ElementType.FIELD})
36  public @interface PersistentForeignKey {
37     /***
38      * The name of the foreign table to map to may be specified,
39      * otherwise the table is determined by looking at the type of the column this
40      * annotation is on and finding the persistent class and therefore the table name for it
41      */
42     String foreignTableName() default "";
43     /***
44      * The name of the column in the foreign table to map to the column referred to by this annotation,
45      * otherwise this is determined by looking at the type of the column this
46      * annotation is on and finding the persistent class and therefore the persistent id for it
47      */
48     String foreignIdColumn() default "";
49  }
