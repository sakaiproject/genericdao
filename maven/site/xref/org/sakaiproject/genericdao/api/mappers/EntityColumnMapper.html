15  package org.sakaiproject.genericdao.api.mappers;
16  
17  import java.util.Map;
18  
19  /***
20   * Extension for the DataMapper which allows custom translation of 
21   * persistent entity data into data maps and back
22   * 
23   * @author Aaron Zeckoski (azeckoski@gmail.com)
24   */
25  public interface EntityColumnMapper extends DataMapper {
26  
27     /***
28      * Called when the object is being input into the database,
29      * this translates an object type to be persisted into a map of column names and values
30      * 
31      * @param persistentObject a persistent object
32      * @return a map of database column names to the values to place in those columns
33      * OR return null to have generic dao attempt to handle this automatically
34      */
35     public Map<String, Object> mapObjectToColumns(Object persistentObject);
36  
37     /***
38      * Called when the object is being pulled out of the database,
39      * this translates a map with column names and the data contained into a persistent object type
40      * @param columnsData a map of database column names to the values in those columns
41      * @return a persistent object with the data from the columnsdata placed in it
42      * OR return null to have generic dao attempt to handle this automatically
43      */
44     public Object mapColumnsToObject(Map<String, Object> columnsData);
45  
46  }
